📋 Step 2 — Add this server route to Forge Builder
// POST /api/moon-chat — proxy to Town Square Moons (builder powerhouse)
app.post("/api/moon-chat", async (req, res) => {
  try {
    const { userId, moon, messages } = req.body;
    if (!userId || !moon || !Array.isArray(messages)) {
      return res.status(400).json({ error: "userId, moon, messages required" });
    }
    // Only the five moons authorized for this app
    if (!["forge", "sage", "quill", "hawk", "creed"].includes(moon)) {
      return res.status(400).json({
        error: "Forge Builder only supports Forge, Sage, Quill, Hawk, and Creed"
      });
    }
    const response = await fetch(`${process.env.TSQ_BASE_URL}/api/moon/chat`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "x-moon-api-key": process.env.TSQ_MOON_API_KEY,
      },
      body: JSON.stringify({ userId, moonName: moon, messages }),
    });
    const data = await response.json();
    res.status(response.status).json(data);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});
